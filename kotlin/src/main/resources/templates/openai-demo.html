// Chat Completion Implementation
suspend fun chatCompletion(messages: List<Message>): ChatResponse {
    val requestBody = ChatRequest(
        model = "gpt-4",
        messages = messages,
        temperature = 0.7
    )
    return httpClient.post("$baseUrl/chat/completions") {
        headers { append("Authorization", "Bearer $apiKey") }
        setBody(requestBody)
    }.body()
}

// Embedding Generation Implementation
suspend fun createEmbedding(text: String): EmbeddingResponse {
    val request = EmbeddingRequest(
        input = text,
        model = "text-embedding-3-small"
    )
    return httpClient.post("$baseUrl/embeddings") {
        headers { append("Authorization", "Bearer $apiKey") }
        setBody(request)
    }.body()
}
