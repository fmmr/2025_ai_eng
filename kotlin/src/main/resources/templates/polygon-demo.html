Code Example

// Polygon Stock Data Integration
val polygon = Polygon(apiKey = "your-polygon-api-key")

// Get aggregates for default tickers (MSFT, NHYDY, TSLA) - last 3 days
val aggregates = polygon.getAggregates()

// Get specific ticker data
val msftData = polygon.getAggregates("MSFT")

// Get company details
val tickerDetails = polygon.getTickerDetails("AAPL")

// Get latest quote
val quote = polygon.getLastQuote("GOOGL")

// Convert to JSON for AI processing
val jsonData = polygon.aggregatesToJson(aggregates)

polygon.close()
